### 基础用法

:::demo
```html
<template>
  <xdh-map>
    <xdh-map-text text="文字内容" :position="[120, 30]" font="40px"></xdh-map-text>

</template>

<script>
  import {XdhMap, XdhMapText} from 'xdh-map'

    components: {
      XdhMap,
      XdhMapText
    }
  }
</script>
```
